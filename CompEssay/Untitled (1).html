Bouncing Balls Comp Essay¶

In this compEssay I will be simulating the movement of billiard balls on a square table. The balls are electrically charged and so will produce an electric and magnetic field (when moving) which will affect their movement. The point is to see how increasing or decreasing the charge changes their behaviour.

# setting up

import numpy as np
import matplotlib.pyplot as pl
import matplotlib.animation as animation


# the constants
eps_0 = 8.854e-12
mu_0 = 1.256e-6
forPie = 4 * np.pi
k = 20000000

# the boundaries

# 3 meters in the negative and positive direction
xlim = 3
ylim = 3

# table surface z coord, unlimited in pos z-direction
zlim = 0

# ball class for keeping track of everything


class Ball:
    def __init__(self, x, y, z, c, R, m, V):

        """
        x, y, z = position
        c = charge
        R = radius of the ball
        m = mass

        """

        self.P = np.array([x, y, z])

        self.R = R
        self.c = c
        self.m = m

        # velocity
        self.V = V

        # current total force acting on the ball
        # will be reset after each loop
        self.F = np.zeros(3)


# and a list of all the balls in play
balls = []

Collisions between balls¶

Collisions between balls will be simulated as a spring compression. I will be using this equation which supposedly realistically approximates the forces in such a collision.

|F| = k * deltaR**(3/2) # this is the force both balls experience

where deltaR is the distance between balls and k = 20_000_000. This force only activates when they are in contact.

# calculates the spring force of all balls that are currently touching

def calcColF():

    N = len(balls)

    for i in range(N - 1):
        for j in range(i + 1, N):

            # F = force from j on i,
            # then the force from i on j is -F
            # so there is no need to calculate it twice

            r = balls[i].P - balls[j].P
            rnorm = np.linalg.norm(r)
            dist = balls[i].R + balls[j].R
            if rnorm < dist:  # if they are touching

                F = k * (dist - rnorm) ** 1.5 * r / rnorm

                balls[i].F += F  # ball i gets hit in the direction of r
                balls[j].F -= F  # ball j in the opposite direction

Collisions with the table¶

The table has boundaries in the x, y and negative z direction. And collisions with these will, again, be simulated as a spring compression, with

|F| = k * deltaR

k being the same as above and deltaR the distance to the edge

# table collisions, four walls and table surface

def calcTableF():

    for b in balls:

        P = b.P

        # check x-coord limits
        if P[0] - b.R < -xlim:

            # distance between ball surface and table wall
            dist = abs(P[0] - b.R + xlim)
            b.F += 10000 * dist * np.array([1, 0, 0])  # in the positive x-direction
        else:
            if P[0] + b.R > xlim:

                dist = abs(P[0] + b.R - xlim)
                b.F += 10000 * dist * np.array([-1, 0, 0])

        # check y-coord limits
        if P[1] - b.R < -ylim:

            dist = abs(P[1] - b.R + ylim)
            b.F += 1000 * dist * np.array([0, 1, 0])
        else:
            if P[1] + b.R > ylim:

                dist = abs(P[1] + b.R - ylim)
                b.F += 1000 * dist * np.array([0, -1, 0])

        # check z-coord limits
        if P[2] - b.R < -zlim:

            dist = abs(P[2] - b.R + zlim)
            b.F += 1000 * dist * np.array([0, 0, 1])

def calcEf():

    """
    calculates the forces on all the balls
    that come from the electric field

    """
    N = len(balls)

    for i in range(N - 1):

        for j in range(i + 1, N):

            # F = force from j on i,
            # then the force from i on j is -F
            # so there is no need to calculate it twice

            r = balls[i].P - balls[j].P
            F = balls[i].c * balls[j].c * r / (forPie * eps_0 * np.linalg.norm(r) ** 3)
            balls[i].F += F
            balls[j].F -= F
            

def calcBf():

    """
    calculates the forces on all the balls
    that come from the magnetic fields

    """

    # since the magnetic field is made of magic and doesnt follow N3L
    # it needs to be calculated from (every ball) on (every ball)

    # from
    for b1 in balls:

        # to
        for b2 in balls:
            if b1 != b2:

                r = b2.P - b1.P
                B = mu_0 / forPie * (np.cross(b1.c * b1.V, r)) / np.linalg.norm(r) ** 3
                b2.F += np.cross(b2.c * b2.V, B)

    B = np.zeros(3)

from IPython.display import HTML

fig = pl.figure()
ax = pl.axes(xlim=(-xlim, xlim), ylim=(-ylim, ylim))

# time step, want to simulate for 3 seconds
dt = 0.001

N = int(3 / dt)  # the number of frames

radius = 0.1  # of a ball
charge = 1e-4

# making some balls

balls.append(Ball(0, 0, radius, charge, radius, 1, np.array([1, 0, 0]))) # give it an initial velocity
balls.append(Ball(1, 1, radius, charge, radius*2, 1, np.array([0, 0, 0])))
balls.append(Ball(-1, -1, radius, charge, radius*3, 1, np.array([0, 0, 0])))

art = []
def init_anim():

    for b in balls:
        circ = pl.Circle((b.P[0], b.P[1]), b.R)
        ax.add_patch(circ)
        art.append(circ)

    return art


# the main calculation function
def animate(i):

    # reset the forces
    for b in balls:
        b.F = np.zeros(3)

    # calculate all the forces
    calcEf()  # electric
    calcBf()  # magnetic
    calcColF()  # collision with balls
    calcTableF()  # collision with table

    for i in range(len(balls)):  # using euler-cromer
        b = balls[i]
        b.V = b.V + dt * b.F / b.m
        b.P = b.P + dt * b.V
        art[i].center = (b.P[0], b.P[1])

    return art


# interval = 1, 1ms between frames = dt
anim = animation.FuncAnimation(
    fig, animate, init_func=init_anim, frames=N, interval=1, blit=False
)

HTML(anim.to_html5_video())
